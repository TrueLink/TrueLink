2. '> publish_modules'
3. '> npm ddp' (optional)
4. '> cd frontend'
5. install_modules
6. '> npm install'
7. '> npm ddp' (optional)
8. '> gulp'
9. '> start.bat'
10. go to http://127.0.0.1:8010/build/
